471 template <typename ConditionType, typename... Args>
472 std::string NewConditionHelper( bool active, Args&&... args )
473 {
474 unsigned number = createConditionNumber();
475 std::unique_ptr<ConditionType> condition(
476 new ConditionType( this, number, active, std::forward<Args>( args )... ) );
477 std::string path = condition->getObjectPath();
478 if ( container )
479 {
480 std::unique_ptr<typename ConditionType::ServiceType> service( new typename ConditionType::ServiceType(
481 condition.get(), std::bind( &ActionSink::removeCondition, this, condition.get() ) ) );
482 service->set_owner( container->get_calling_client_id() );
483 condition->set_service( service.get() );
484 container->create_object( path, std::move( service ) );
485 }
486 conditions.insert( std::make_pair( number, std::move( condition ) ) );
487 if ( active )
488 {
489 eca.compile();
490 }
491 return path;
492 }
