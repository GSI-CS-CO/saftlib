488 template<typename ConditionType, typename... Args>
489 std::string NewConditionHelper(bool active, Args&&... args) {
490 unsigned number = createConditionNumber();
491 std::unique_ptr<ConditionType> condition(new ConditionType(this, number, active, std::forward<Args>(args)...));
492 std::string path = condition->getObjectPath();
493 if (container) {
494 std::unique_ptr<typename ConditionType::ServiceType> service(new typename ConditionType::ServiceType(condition.get(), std::bind(&ActionSink::removeCondition, this, condition.get())));
495 service->set_owner(container->get_calling_client_id());
496 condition->set_service(service.get());
497 container->create_object(path, std::move(service));
498 }
499 conditions.insert(std::make_pair(number, std::move(condition)));
500 if (active) {
501 eca.compile();
502 }
503 return path;
504 }
