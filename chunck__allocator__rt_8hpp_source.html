30class ChunckAllocatorRT {
31public:
32 ChunckAllocatorRT()
33 : allocated_chuncks(0)
34 {
35 for (size_t i = 0; i < MAX_CHUNCKS; ++i) {
36 chuncks[i].index = i;
37 indices[i] = i;
38 }
39 }
40 char* malloc(size_t size) {
41 assert(size <= CHUNCKSIZE);
42 assert(allocated_chuncks < MAX_CHUNCKS);
43 return chuncks[indices[allocated_chuncks++]].buffer;
44 }
45
46 void free(char* ptr) {
47 assert(allocated_chuncks > 0);
48 assert(contains(ptr));
49 size_t index = (ptr-chuncks[0].buffer)/sizeof(Chunck);
50
51 size_t &a_idx = indices[chuncks[index].index];
52 size_t &b_idx = indices[allocated_chuncks-1];
53
54 size_t &a_backidx = chuncks[a_idx].index;
55 size_t &b_backidx = chuncks[b_idx].index;
56
57 // std::cerr << "swapping " << a_idx << " <-> " << b_idx << std::endl;
58 std::swap(a_idx, b_idx);
59 // std::cerr << "swapping " << a_backidx << " <-> " << b_backidx << std::endl;
60 std::swap(a_backidx, b_backidx);
61
62 --allocated_chuncks;
63 }
64 void print_size() {
65 std::cerr << "filled: " << allocated_chuncks << "/" << MAX_CHUNCKS << std::endl;
66 }

69 for (int i = 0; i < MAX_CHUNCKS; ++i) {
70 std::cerr << std::setw(3) << chuncks[i].index << " ";

81 }
